# PureLeaf — Pretty URL Rewrite Rules (Apache / cPanel)
# Rename to .htaccess and upload to site root.

Options -Indexes
RewriteEngine On
RewriteRule ^blog/([a-z0-9-]+)/?$  post.html?slug=$1  [L,QSA]
RewriteRule ^blog/?$  blog.html  [L]
RewriteRule ^about/?$  about.html  [L]
RewriteRule ^home/?$  index.html  [L]
